# Implement a tri-gram model

import nltk
nltk.download('punkt')

string = "i am your best friend always and forever"
qt=nltk.word_tokenize(string)
print(qt)
print("---------------------------")

qu=list(nltk.bigrams(qt))
print(qu)
print("---------------------------")

tri=list(nltk.ngrams(qt,3))
print(tri)
